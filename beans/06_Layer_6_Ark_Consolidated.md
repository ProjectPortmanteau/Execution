        ARRAY[b.bean_id] AS path
    FROM ark_beans b
    LEFT JOIN ark_edges e ON e.source_id = b.bean_id
    WHERE b.bean_id = 'PHIL-001'  -- << Change start point here

    SELECT
        nb.bean_id,
        nb.layer,
        nb.title,
        nb.category,
        ne.edge_type,
        bw.depth + 1,
        bw.path || nb.bean_id

    JOIN ark_edges ce ON ce.source_id = bw.bean_id
    JOIN ark_beans nb ON nb.bean_id = ce.target_id
    LEFT JOIN ark_edges ne ON ne.source_id = nb.bean_id
    WHERE nb.bean_id <> ALL(bw.path)  -- Prevent cycles

)
